```python
import os
import re
import shutil

# 配置项
BLOG_ROOT = r"D:/Code/blog/content/post"  # 你的博客根目录
BACKUP_ROOT_DIR = os.path.join(BLOG_ROOT, "backup_before_rename")  # 备份目录
IMAGE_EXTENSIONS = (".png", ".jpg", ".jpeg", ".gif", ".webp")  # 支持的图片格式


def create_backup(src_path):
    """创建文件/文件夹备份"""
    if not os.path.exists(BACKUP_ROOT_DIR):
        os.makedirs(BACKUP_ROOT_DIR)

    rel_path = os.path.relpath(src_path, BLOG_ROOT)
    backup_file_path = os.path.join(BACKUP_ROOT_DIR, rel_path)

    if os.path.isdir(src_path):
        # 如果是文件夹，递归复制（本脚本中基本用不到，预留扩展）
        if not os.path.exists(backup_file_path):
            shutil.copytree(src_path, backup_file_path)
    elif os.path.isfile(src_path):
        # 如果是文件：先创建备份目录的父文件夹 → 再复制文件
        backup_file_dir = os.path.dirname(backup_file_path)
        if not os.path.exists(backup_file_dir):
            os.makedirs(backup_file_dir)
        shutil.copy2(src_path, backup_file_path)


def parse_front_matter(content):
    """解析FrontMatter内容，返回字典"""
    fm_pattern = re.compile(r"^---\n(.*?)\n---", re.DOTALL)
    match = fm_pattern.match(content)
    if not match:
        return {}, content

    fm_content = match.group(1).strip()
    fm_dict = {}
    current_key = None  # 跟踪当前正在解析的列表型键
    lines = [line.strip() for line in fm_content.split("\n")]

    for line in lines:
        # 跳过空行/注释行
        if not line or line.startswith("#"):
            continue
        line_stripped = line.strip()

        # 场景1：匹配键（key: 格式，无论后面有没有值）
        if line_stripped.endswith(":") and not line_stripped.startswith("-"):
            # 提取键名（去掉末尾的:）
            key = line_stripped[:-1].strip()
            # 标记当前键为待解析列表，初始化空列表
            fm_dict[key] = []
            current_key = key
        # 场景2：普通键值对（key: value 格式）
        elif ":" in line_stripped and not line_stripped.startswith("-"):
            key, value = line_stripped.split(":", 1)
            key = key.strip()
            value = value.strip()

            fm_dict[key] = value
            current_key = None  # 重置列表标记
        # 场景3：延续上一个列表键（比如 tags 下的 - java）
        elif line_stripped.startswith("-") and current_key is not None:
            list_item = line_stripped.lstrip("-").strip()
            if list_item:  # 避免空项
                fm_dict[current_key].append(list_item)

    # 提取正文并标准化代码块语言
    raw_main_content = content[match.end() :].strip()
    main_content = standardize_code_block_lang(raw_main_content)
    return fm_dict, main_content

def standardize_code_block_lang(text):
    """
    标准化代码块的语言标识，适配astro-expressive-code
    比如：JAVA→java、JS→js、PYTHON→python 等
    """
    # 1. 定义语言标识映射表
    lang_mapping = {
        "JS": "js",
        "JAVASCRIPT": "js",
        "TS": "ts",
        "TYPESCRIPT": "ts",
        "HTML": "html",
        "CSS": "css",
        "SCSS": "scss",
        "LESS": "less",
        "JAVA": "java",
        "PYTHON": "python",
        "PY": "py",
        "GO": "go",
        "RUST": "rust",
        "PHP": "php",
        "CPP": "cpp",
        "C++": "cpp",
        "C": "c",
        "JSON": "json",
        "YAML": "yaml",
        "MD": "md",
        "MARKDOWN": "md",
        "SH": "sh",
        "BASH": "bash",
    }
    # 2. 正则匹配代码块（```xxx 开头的行）
    code_block_pattern = re.compile(r"^```(\w+)", re.MULTILINE)
    # 3. 替换函数：将不规范的语言标识转为标准格式
    def replace_lang(match):
        original_lang = match.group(1).upper()  # 统一转大写匹配
        # 查找映射，无匹配则保留原标识（小写）
        standard_lang = lang_mapping.get(original_lang, original_lang.lower())
        return f"```{standard_lang}"
    # 4. 执行替换
    standardized_text = code_block_pattern.sub(replace_lang, text)
    return standardized_text


def generate_new_front_matter(fm_dict, folder_name):
    """生成新格式的FrontMatter"""
    # 1. 处理标题（优先用原title，无则用文件夹名）
    title = fm_dict.get("title", folder_name)

    # 2. 处理发布日期
    raw_date = fm_dict.get("date", "")
    published = ""
    if raw_date:
        # 优先从文件夹名提取日期（文件夹名格式：25-04-20-xxx）
        date_match = re.match(r"(\d{2})-(\d{2})-(\d{2})", folder_name)
        if date_match:
            year = f"20{date_match.group(1)}"
            month = date_match.group(2)
            day = date_match.group(3)
            published = f"{year}-{month}-{day}"
        else:
            # 用原data字段，补全年份
            published = raw_date

    # 3. 处理描述
    description = fm_dict.get("description", title)
    # 4. 处理标签（原tags → 转为逗号分隔的列表）
    tags = fm_dict.get("tags", [])
    tags_str = ", ".join(tags) if tags else "Uncategorized"
    # 5. 处理分类（原categories取第一个 → category）
    categories = fm_dict.get("categories", [])
    category = categories[0] if categories else "Uncategorized"

    # 拼接新的FrontMatter
    new_fm = f"""---
title: {title}
published: {published}
description: {description}
tags: [{tags_str}]
category: {category}
---"""
    return new_fm


def process_image_references(content, folder_path, folder_name):
    """处理图片引用：重命名图片文件 + 更新MD中的引用路径"""
    # 提取所有图片引用（![xxx](xxx.png) 格式）
    img_pattern = re.compile(
        r"!\[.*?\]\((.*?\.(?:png|jpg|jpeg|gif|webp))\)", re.IGNORECASE
    )
    img_matches = img_pattern.findall(content)
    if not img_matches:
        return content  # 无图片引用，直接返回

    # 按引用顺序重命名图片
    updated_content = content
    for idx, img_path in enumerate(img_matches, 1):
        # 获取图片文件名（如 2025-05-27-21-24-06.png）
        img_filename = os.path.basename(img_path)
        old_img_full_path = os.path.join(folder_path, img_filename)

        if not os.path.exists(old_img_full_path):
            print(f"警告：图片文件不存在 {old_img_full_path}，跳过")
            continue

        # 新文件名：文件夹名_序号.扩展名
        img_ext = os.path.splitext(img_filename)[1]
        new_img_filename = f"{folder_name}_{idx}{img_ext}"
        new_img_full_path = os.path.join(folder_path, new_img_filename)

        # 重命名文件（先备份）
        create_backup(old_img_full_path)
        os.rename(old_img_full_path, new_img_full_path)
        print(f"重命名图片：{old_img_full_path} → {new_img_full_path}")

        # 更新MD中的图片引用
        updated_content = updated_content.replace(img_path, new_img_filename)

    return updated_content


def process_single_post(folder_path):
    """处理单个博客文章文件夹"""
    # 获取文件夹名（用于标题/图片命名）
    folder_name = os.path.basename(folder_path)

    # 定位index.md文件（兼容非index.md的情况）
    md_files = [f for f in os.listdir(folder_path) if f.endswith(".md")]
    if not md_files:
        print(f"警告：{folder_path} 中未找到MD文件，跳过")
        return
    md_file_path = os.path.join(folder_path, md_files[0])

    # 读取原文件内容（先备份）
    create_backup(md_file_path)
    with open(md_file_path, "r", encoding="utf-8") as f:
        content = f.read()

    # 1. 解析并重构FrontMatter
    fm_dict, main_content = parse_front_matter(content)
    new_fm = generate_new_front_matter(fm_dict, folder_name)

    # 2. 处理图片引用和重命名
    updated_main_content = process_image_references(
        main_content, folder_path, folder_name
    )

    # 3. 拼接新内容并写入文件
    new_content = f"{new_fm}\n\n{updated_main_content}"
    with open(md_file_path, "w", encoding="utf-8") as f:
        f.write(new_content)

    print(f"成功处理：{md_file_path}")


def main():
    """主函数：遍历所有博客文件夹并处理"""
    print(f"开始处理博客目录：{BLOG_ROOT}")
    print(f"备份目录：{BACKUP_ROOT_DIR}")
    print("=" * 50)

    # 遍历所有子文件夹
    for item in os.listdir(BLOG_ROOT):
        item_path = os.path.join(BLOG_ROOT, item)
        if os.path.isdir(item_path) and not item.startswith("_"):  # 跳过特殊文件夹
            process_single_post(item_path)

    print("=" * 50)
    print("处理完成！")
    print(f"注意：所有原始文件已备份至 {BACKUP_ROOT_DIR}")


if __name__ == "__main__":
    # 安全确认
    confirm = input("执行此脚本会修改文件并自动备份，是否继续？(y/n)：")
    if confirm.lower() == "y":
        main()
    else:
        print("操作已取消")

import shutil
import os

file_path = os.path.abspath(__file__)
current_dir = os.path.split(file_path)[0]
print(f"开始处理当前目录:{current_dir}")

for entry in os.listdir(current_dir):
    entry_path = os.path.join(current_dir, entry)
    if os.path.isdir(entry_path) and not entry.startswith("."):
        folder_contents = [f for f in os.listdir(entry_path) if not f.startswith(".")]
        if len(folder_contents) == 1 and folder_contents[0] == "index.md":
            # 拼接原文件路径和新文件路径
            original_file_path = os.path.join(entry_path, "index.md")
            new_file_path = os.path.join(current_dir, f"{entry}.md")
            if os.path.exists(new_file_path):
                print(f"文件 {new_file_path} 已存在,跳过重命名。")
                continue
            try:
                # 移动并重命名文件(剪切+重命名一步完成)
                shutil.move(original_file_path, new_file_path)
                print(f" 已将 {original_file_path} 移动并重命名为 {new_file_path}")

                # 删除空文件夹
                os.rmdir(entry_path)
                print(f" 已删除空文件夹:{entry_path}")

            except Exception as e:
                print(f" 处理文件夹 {entry} 时出错:{str(e)}")
        else:
            # 不符合条件的情况说明
            if len(folder_contents) != 1:
                print(
                    f" 文件夹 {entry} 内文件数量不为1(共{len(folder_contents)}个),跳过"
                )
            else:
                print(
                    f" 文件夹 {entry} 内唯一文件不是index.md(是{folder_contents[0]}),跳过"
                )
