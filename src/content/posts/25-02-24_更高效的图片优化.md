### pyvips版本
```python
import os
import pyvips

# 获取当前工作目录
current_dir = os.getcwd()
# 压缩质量（默认80，0-100）
compression_quality = 80
# 支持的图片扩展名
image_extensions = ('.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.tif')

print(f"开始递归转换 {current_dir} 下的所有图片为 webp 格式（替换原文件,压缩质量：{compression_quality}）...\n")

def process_images_recursively(dir_path):
    """递归遍历目录并处理图片"""
    for entry in os.scandir(dir_path):
        # 如果是目录，递归处理
        if entry.is_dir(follow_symlinks=False):
            process_images_recursively(entry.path)
            continue
        
        # 筛选图片文件
        if entry.name.lower().endswith(image_extensions):
            original_path = entry.path
            # webp路径
            webp_path = os.path.splitext(original_path)[0] + '.webp'
            
            try:
                # 使用pyvips读取并转换为webp
                # pyvips自动处理透明通道，无需手动转换
                image = pyvips.Image.new_from_file(original_path, access='sequential')
                # 保存为webp文件
                image.write_to_file(
                    webp_path,
                    format='webp',
                    Q=compression_quality,  # webp压缩质量
                    lossless=False  # 无损压缩设为False，如需无损可改为True
                )
                
                # 计算压缩率
                original_size = os.path.getsize(original_path) / 1024
                new_size = os.path.getsize(webp_path) / 1024
                compression_ratio = (1 - new_size/original_size) * 100
                
                # 删除原文件
                os.remove(original_path)
                
                print(f"替换完成: {original_path}")
                print(f"压缩率: {compression_ratio:.1f}% (原{original_size:.1f}KB → 新{new_size:.1f}KB)\n")
            
            except Exception as e:
                print(f"处理失败 {original_path}: {str(e)}\n")
                # 清理临时文件
                if os.path.exists(webp_path):
                    os.remove(webp_path)

# 执行处理
process_images_recursively(current_dir)
print("所有图片转换替换完成")
```

### sharpjs版本
```js
import { promises as fs } from 'fs';
import { existsSync } from 'fs'; // 用于同步判断文件是否存在
import { join, extname, basename } from 'path';
import sharp from 'sharp';

// 配置项
const compressionQuality = 80; // WebP 压缩质量 (0-100)
const imageExtensions = ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.tif'];
const currentDir = process.cwd(); // 获取当前工作目录

async function processImagesRecursively(dirPath) {
    try {
        // 读取目录下的所有文件/子目录
        const entries = await fs.readdir(dirPath, { withFileTypes: true });

        for (const entry of entries) {
            const originalPath = join(dirPath, entry.name);

            // 如果是目录，递归处理
            if (entry.isDirectory()) {
                await processImagesRecursively(originalPath);
                continue;
            }

            // 筛选图片文件（忽略大小写）
            const ext = extname(entry.name).toLowerCase();
            if (!imageExtensions.includes(ext)) {
                continue;
            }

            // 构建Webp文件路径
            const webpPath = join(
                dirPath,
                basename(entry.name, ext) + '.webp'
            );

            try {
                // 获取原文件大小
                const originalStats = await fs.stat(originalPath);
                const originalSize = originalStats.size / 1024; // 转换为 KB

                // 使用 sharp 处理图片并保存为Webp文件
                await sharp(originalPath)
                    .webp({ 
                        quality: compressionQuality,
                        lossless: false // 无损压缩，如需开启改为 true
                    })
                    .toFile(webpPath);

                // 获取新文件大小
                const newStats = await fs.stat(webpPath);
                const newSize = newStats.size / 1024;
                const compressionRatio = (1 - newSize / originalSize) * 100;

                // 删除原文件
                await fs.unlink(originalPath);

                // 输出处理结果
                console.log(`替换完成: ${originalPath}`);
                console.log(`压缩率: ${compressionRatio.toFixed(1)}% (原${originalSize.toFixed(1)}KB → 新${newSize.toFixed(1)}KB)\n`);

            } catch (processError) {
                console.error(`处理失败 ${originalPath}: ${processError.message}\n`);
                // 清理临时文件
                if (existsSync(webpPath)) {
                    await fs.unlink(webpPath);
                }
            }
        }
    } catch (dirError) {
        console.error(`读取目录失败 ${dirPath}: ${dirError.message}`);
    }
}

// 执行主逻辑
(async () => {
    console.log(`开始递归转换 ${currentDir} 下的所有图片为 webp 格式（替换原文件,压缩质量：${compressionQuality}）...\n`);
    await processImagesRecursively(currentDir);
    console.log("所有图片转换替换完成");
})();
```
