### 1. 核心依赖与前置准备
```typescript
import router from '@/router';
import NProgress from 'nprogress'; // 进度条（可选）
import 'nprogress/nprogress.css';
import { useUserStore } from '@/store/modules/user';
import { usePermissionStore } from '@/store/modules/permission';
import { AuthStorage, redirectToLogin } from '@/utils/auth';
import { ROLE_ROOT } from '@/constants';

NProgress.configure({ showSpinner: false }); // 进度条配置
const whiteList = ['/login', '/401', '/404']; // 无需权限的白名单页面
```

### 2. 路由前置守卫（核心重定向逻辑）
```typescript
router.beforeEach(async (to, from, next) => {
  NProgress.start(); // 启动进度条
  const userStore = useUserStore();
  const permissionStore = usePermissionStore();

  // 1. 判断用户是否已登录（通过 token 校验）
  const hasToken = AuthStorage.getAccessToken();

  if (hasToken) {
    // 场景1：已登录用户访问登录页 → 重定向到首页
    if (to.path === '/login') {
      next({ path: '/' });
      NProgress.done();
    } else {
      // 2. 已登录用户：校验是否有用户信息（避免刷新后丢失）
      const hasUserInfo = !!userStore.userInfo.roles;
      if (hasUserInfo) {
        // 3. 有用户信息：校验页面权限
        if (hasPermission(to, userStore.userInfo.roles)) {
          next(); // 有权限 → 正常访问
        } else {
          // 无权限 → 重定向到 401 页面
          next({ path: '/401', replace: true });
        }
      } else {
        try {
          // 4. 无用户信息：刷新用户信息和动态路由
          await userStore.getUserInfo(); // 获取用户信息（含角色/权限）
          const accessRoutes = await permissionStore.generateRoutes(); // 生成动态路由
          
          // 动态添加路由（解决刷新后动态路由丢失）
          accessRoutes.forEach(route => router.addRoute(route));
          
          // 重定向到目标页（replace: true 避免回退到守卫前的页面）
          next({ ...to, replace: true });
        } catch (error) {
          // 5. 获取用户信息失败（token 过期/无效）→ 清空token + 重定向到登录页
          await userStore.resetToken();
          redirectToLogin(`登录已过期，请重新登录`);
          next({ path: '/login', query: { redirect: to.fullPath } }); // 带来源页参数
          NProgress.done();
        }
      }
    }
  } else {
    // 6. 未登录用户：白名单页面直接访问，否则重定向到登录页
    if (whiteList.includes(to.path)) {
      next();
    } else {
      next({ path: '/login', query: { redirect: to.fullPath } }); // 重定向到登录页，带回跳参数
      NProgress.done();
    }
  }
});

// 路由后置守卫：结束进度条
router.afterEach(() => {
  NProgress.done();
});

// 辅助函数：校验页面权限（匹配路由meta.roles）
function hasPermission(to: RouteLocationNormalized, roles: string[]) {
  if (to.meta && to.meta.roles) {
    // 超级管理员（ROLE_ROOT）拥有所有权限
    if (roles.includes(ROLE_ROOT)) return true;
    // 匹配当前路由需要的角色
    return roles.some(role => to.meta.roles!.includes(role));
  }
  // 无角色要求的页面默认可访问
  return true;
}
```

### 三、关键重定向场景详解
#### 1. 登录页重定向（带回跳参数）
```typescript
next({ path: '/login', query: { redirect: to.fullPath } });
```
- 作用：未登录用户访问需要权限的页面时，重定向到登录页，并在 `query` 中携带 `redirect` 参数（当前访问的页面路径）；
- 登录成功后，可通过 `router.push(route.query.redirect || '/')` 跳回来源页

```typescript
// src/views/login/components/Login.vue
import { useRoute } from "vue-router";

const route = useRoute();
// 读取登录页URL中的redirect参数（即用户原本想访问的页面）
const redirect = ref(route.query.redirect || "/"); 

// 登录表单提交方法
const handleLogin = async () => {
  if (!loginFormRef.value) return;
  
  try {
    // 1. 表单校验
    await loginFormRef.value.validate();
    loading.value = true;
    
    // 2. 调用登录接口，获取token等信息
    const res = await AuthAPI.login(loginFormData.value);
    
    // 3. 存储token/用户信息（Pinia + 本地存储）
    await userStore.setToken(res.accessToken); // 存储token到Pinia和localStorage
    await userStore.getUserInfo(); // 获取并存储用户信息
    
    // 4. 核心：根据redirect参数重定向
    // 方式1：简单重定向（直接跳转）
    router.push(redirect.value as string);
    
    // 方式2：复杂路由重定向（处理带参数的redirect，可选）
    // 如果redirect是完整路由对象（如"/detail/1?name=test"），可解析后跳转
    // const redirectPath = decodeURIComponent(redirect.value as string);
    // router.push(redirectPath);
    
    // 5. 提示登录成功
    ElMessage.success(t("login.loginSuccess"));
  } catch (error) {
    // 登录失败：刷新验证码、提示错误
    getCaptcha(); 
    ElMessage.error(t("login.loginFailed"));
  } finally {
    loading.value = false;
  }
};
```

#### 2. 401 权限不足重定向
```typescript
next({ path: '/401', replace: true });
```
- 作用：已登录用户访问无权限的页面时，重定向到 401 页面（对应代码库中 `src/views/error/401.vue`）；
- `replace: true`：替换当前路由记录，避免用户点击回退按钮回到无权限页面。

#### 3. 动态路由生成后的重定向
```typescript
next({ ...to, replace: true });
```
- 场景：用户刷新页面后，动态路由丢失，需要重新生成动态路由；
- 作用：重新触发路由守卫，确保动态路由加载完成后跳转到目标页（避免直接 `next()` 导致路由匹配失败）。

#### 4. 登录后重定向到首页
```typescript
if (to.path === '/login') {
  next({ path: '/' });
}
```
- 作用：已登录用户再次访问登录页时，自动重定向到首页（避免重复登录）。

### 四、重定向细节
1. **redirect 页面**：
```vue
// src/views/redirect/index.vue
<script setup lang="ts">
import { useRoute, useRouter } from "vue-router";
const route = useRoute();
const router = useRouter();
const { params, query } = route;
const { path } = params;
router.replace({ path: "/" + path, query }); // 解析 redirect 参数并重定向
</script>
```
- 作用：解决复杂路由的重定向参数传递（如嵌套路由、带 query 参数的路由），避免参数丢失（对应 CHANGELOG 中修复的“注销登出后 redirect 跳转路由参数丢失”问题）。

2. **401 页面的回退/首页跳转** ：
```vue
// src/views/redirect/index.vue
<el-button icon="arrow-left" @click="router.back()">返回</el-button>
<el-link type="primary" @click="router.push('/dashboard')">回首页</el-link>
```
- 补充：权限不足页面提供手动重定向选项，提升用户体验。
