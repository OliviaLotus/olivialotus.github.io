# 一、系统与路径：os / sys / pathlib

## 1. os 操作系统接口

```python
import os

# 1. 获取当前工作目录
cwd = os.getcwd()
print("当前工作目录:", cwd)
# 输出类似: 当前工作目录: /Users/xxx/Desktop/python_demo

# 2. 列出当前目录下的文件和文件夹
files = os.listdir('.')
print("当前目录内容:", files)
# 输出: 当前目录内容: ['test.py', 'data', 'demo.txt']

# 3. 判断是否是文件 / 文件夹
print("是否是文件:", os.path.isfile('demo.txt'))   # 输出: True 或 False
print("是否是文件夹:", os.path.isdir('data'))      # 输出: True 或 False

# 4. 拼接路径
path = os.path.join('folder', 'sub', 'file.txt')
print("拼接后的路径:", path)
# 输出: folder/sub/file.txt或 folder\sub\file.txt

# 5. 获取文件大小
if os.path.exists('demo.txt'):
    print("文件大小:", os.path.getsize('demo.txt'), "字节")

# 6. 创建 / 删除 文件夹
os.mkdir('new_dir')    # 创建单层目录
os.rmdir('new_dir')    # 删除空目录
```

## 2. sys 解释器相关

```python
import sys

# 1. 获取命令行参数
# 运行: python test.py a b c
print("命令行参数:", sys.argv)
# 输出: ['test.py', 'a', 'b', 'c']
# sys.argv[0] = 脚本文件名
# sys.argv[1] = 第一个参数

# 2. 获取 Python 版本
print("Python版本:", sys.version)
# 输出: 3.11.4 (main, ...)

# 3. 退出程序

# 运行到这里会直接终止，并打印: 程序主动退出

# 4. 查看模块搜索路径
print("Python 找模块的路径:")
for p in sys.path:
    print(p)
# 输出一堆路径，Python 就是从这里 import 模块
```

## 3. pathlib

### 一、基础：创建 Path 对象（核心入口）

`Path` 是 `pathlib` 的核心类，所有操作都从创建路径对象开始。

```python
from pathlib import Path

# ========== 1. 创建绝对路径 ==========
# Windows: C:\Users\你的用户名\test.txt
# Linux/Mac: /home/你的用户名/test.txt
abs_path = Path.home() / "test.txt"
print(abs_path)  # 输出：系统对应的绝对路径（如 PosixPath('/home/user/test.txt')）

# ========== 2. 创建当前目录的相对路径 ==========
# 用 / 拼接路径（最直观的写法，替代 os.path.join）
rel_path = Path(".") / "docs" / "README.md"
print(rel_path)  # 输出：./docs/README.md（或 PosixPath('./docs/README.md')）

# ========== 3. 直接传入完整路径字符串 ==========
str_path = Path("/usr/local/bin/python3")
print(str_path)  # 输出：/usr/local/bin/python3
```

### 二、路径属性（获取路径的各类信息）

这是 `pathlib` 最常用的功能，相比 `os.path` 的零散函数，属性调用更直观。

```python
# 先定义一个示例路径（假设文件存在）
file_path = Path("/home/user/projects/demo/main.py")

# ========== 1. 核心属性 ==========
print(file_path.name)        # 输出：main.py → 带扩展名的完整文件名
print(file_path.stem)        # 输出：main → 纯文件名（无扩展名）
print(file_path.suffix)      # 输出：.py → 文件扩展名（带点）
print(file_path.parent)      # 输出：/home/user/projects/demo → 父目录（Path对象）
print(file_path.parents[0])  # 输出：/home/user/projects/demo → 父目录（同parent）
print(file_path.parents[1])  # 输出：/home/user/projects → 祖父目录
print(file_path.parents[2])  # 输出：/home/user → 曾祖父目录
print(file_path.root)        # 输出：/ → 根目录（Linux/Mac）；Windows下是 C:\ 等

# ========== 2. 路径格式转换 ==========
print(str(file_path))        # 输出：/home/user/projects/demo/main.py → 转为字符串（兼容os.path）
print(file_path.as_posix())  # 输出：/home/user/projects/demo/main.py → 转为POSIX格式路径（统一用/）
print(file_path.as_uri())    # 输出：file:///home/user/projects/demo/main.py → 转为文件URI
```

### 三、路径操作（拼接、修改、解析路径）

```python
base_path = Path("/home/user/docs")

# ========== 1. 拼接路径（推荐用 /，也可以用 joinpath） ==========
new_path = base_path / "notes" / "todo.txt"
# 等价于 base_path.joinpath("notes", "todo.txt")
print(new_path)  # 输出：/home/user/docs/notes/todo.txt

# ========== 2. 修改路径（返回新Path对象，原对象不变） ==========
file_path = Path("report.pdf")
# 替换扩展名
print(file_path.with_suffix(".docx"))  # 输出：report.docx
# 替换文件名（保留扩展名）
print(file_path.with_name("annual_report.pdf"))  # 输出：annual_report.pdf

# ========== 3. 解析相对路径 → 绝对路径 ==========
rel_path = Path("./demo/test.py")
print(rel_path.resolve())  # 输出：/home/user/当前目录/demo/test.py → 解析为绝对路径（自动处理../等）
```

### 四、文件/目录操作（创建、删除、读写、遍历）

`pathlib` 内置了常用的文件/目录操作，无需额外导入 `os`/`open`。

```python
# 先定义测试路径
test_dir = Path("test_dir/sub_dir")
test_file = test_dir / "test.txt"

# ========== 1. 目录操作 ==========
# 创建目录（parents=True：递归创建父目录；exist_ok=True：目录已存在时不报错）
test_dir.mkdir(parents=True, exist_ok=True)
# 执行效果：创建 test_dir/ 和 test_dir/sub_dir/ 目录（如果不存在）

# 遍历目录下的所有文件（返回生成器，可迭代）
for file in Path(".").iterdir():
    print(file)  # 输出：当前目录下的所有文件/目录（如 ./test_dir, ./main.py 等）

# 按模式匹配文件（替代 glob.glob，返回生成器）
# 匹配当前目录及子目录下所有 .py 文件
py_files = Path(".").glob("**/*.py")
print(list(py_files))  # 输出：[PosixPath('./main.py'), PosixPath('./test_dir/sub_dir/test.py')]

# ========== 2. 文件操作 ==========
# 写入文件内容（替代 open + write，自动处理文件打开/关闭）
test_file.write_text("Hello Pathlib!\n这是测试内容", encoding="utf-8")
# 执行效果：在 test_dir/sub_dir/ 下创建 test.txt，写入指定内容

# 读取文件内容（替代 open + read）
content = test_file.read_text(encoding="utf-8")
print(content)  # 输出：Hello Pathlib! 换行 这是测试内容

# 复制文件（需要结合 shutil，pathlib 未内置复制，示例如下）
import shutil
copy_file = test_dir / "test_copy.txt"
shutil.copy(test_file, copy_file)
# 执行效果：创建 test_copy.txt，内容和 test.txt 一致

# ========== 3. 删除操作 ==========
# 删除文件（文件不存在会报错，可先判断）
if copy_file.exists():
    copy_file.unlink()  # 执行效果：删除 test_copy.txt

# 删除目录（空目录才能删；如果要删非空目录，需用 shutil.rmtree）
# 先删除目录内的文件
test_file.unlink()
# 再删除空目录（可递归删除多级空目录）
test_dir.rmdir()  # 执行效果：删除 sub_dir/；test_dir/ 需单独 rmdir
Path("test_dir").rmdir()
```

### 五、路径判断（检查路径状态）

用于判断路径是否存在、是文件还是目录等，方法名语义化极强。

```python
test_path = Path("main.py")
non_exist_path = Path("non_exist.txt")

print(test_path.exists())    # 输出：True/False → 路径是否存在
print(test_path.is_file())   # 输出：True/False → 是否是文件
print(test_path.is_dir())    # 输出：False/True → 是否是目录
print(test_path.is_absolute())  # 输出：False → 是否是绝对路径
print(non_exist_path.exists())  # 输出：False → 不存在的路径返回False

# 额外实用判断
print(test_path.is_symlink())  # 输出：False → 是否是符号链接
print(Path("/").is_mount())    # 输出：True → 是否是挂载点（Linux/Mac）
```

# 二、日期时间：datetime / time

## 1. datetime

```python
from datetime import datetime

# 1. 获取当前时间
now = datetime.now()
print("当前时间:", now)
# 输出: 2026-02-25 15:30:45.123456

# 2. 格式化时间
str_time = now.strftime("%Y-%m-%d %H:%M:%S")
print("格式化后:", str_time)
# 输出: 2026-02-25 15:30:45

# 3. 字符串转时间对象
dt = datetime.strptime("2026-01-01", "%Y-%m-%d")
print("转成时间对象:", dt)
# 输出: 2026-01-01 00:00:00

# 4. 获取年、月、日、时、分、秒
print("年:", now.year)
print("月:", now.month)
print("日:", now.day)
# 2026、2、25
```

## 2. time

```python
import time

# 1. 获取时间戳
ts = time.time()
print("时间戳:", ts)
# 输出: 1740460000.123456

# 2. 程序暂停 2 秒
print("开始等待...")
time.sleep(2)
print("等待结束")
# 先打印“开始等待”，停顿2秒，再打印“等待结束”
```

# 三、数据序列化：json / pickle

## 1. json

```python
import json

data = {
    "name": "张三",
    "age": 20,
    "skills": ["Python", "Java"]
}

# 1. 字典 → JSON字符串
# ensure_ascii=False 保证中文不乱码
json_str = json.dumps(data, ensure_ascii=False, indent=4)
print("JSON字符串:")
print(json_str)
# 输出格式化好的 JSON，中文正常显示

# 2. JSON字符串 → 字典
data2 = json.loads(json_str)
print("解析后名字:", data2["name"])  # 输出: 张三

# 3. 直接写入文件
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=4)
# 运行后：当前目录多出 data.json 文件
```

## 2. pickle

```python
import pickle

class Person:
    def __init__(self, name):
        self.name = name

p = Person("李四")

# 保存对象到文件
with open("person.pkl", "wb") as f:
    pickle.dump(p, f)

# 从文件读取对象
with open("person.pkl", "rb") as f:
    p2 = pickle.load(f)
print("读取的名字:", p2.name)  # 输出: 李四
```

# 五、正则表达式：re

```python
import re

text = "我的电话是 13812345678，邮箱: test@example.com"

# 匹配手机号
phone = re.search(r'1[3-9]\d{9}', text)
if phone:
    print("手机号:", phone.group())  # 输出: 13812345678

# 匹配邮箱
email = re.search(r'\w+@\w+\.\w+', text)
if email:
    print("邮箱:", email.group())    # 输出: test@example.com

# 替换字符串
new_text = re.sub(r'138\d{6}', '138******', text)
print("脱敏后:", new_text)
# 输出: 我的电话是 138******5678，邮箱: test@example.com
```

# 六、并发：concurrent.futures

```python
from concurrent.futures import ThreadPoolExecutor
import time

def task(name):
    print(f"任务 {name} 开始")
    time.sleep(1)
    print(f"任务 {name} 结束")
    return f"结果{name}"

# 多线程执行
with ThreadPoolExecutor(max_workers=3) as executor:
    futures = [executor.submit(task, i) for i in range(3)]
    for f in futures:
        print(f.result())
# 输出：3个任务几乎同时开始，1秒后几乎同时结束，比串行快很多
```

# 九、数学与随机：math / random

```python
import math
import random

# math
print("圆周率:", math.pi)          # 输出: 3.14159...
print("平方根:", math.sqrt(16))    # 输出: 4.0
print("向上取整:", math.ceil(3.1)) # 输出: 4

# random
print("0~1随机数:", random.random())
print("1~10整数:", random.randint(1, 10))

lst = [1,2,3,4]
random.shuffle(lst)  # 打乱列表
print("打乱后:", lst)
```
