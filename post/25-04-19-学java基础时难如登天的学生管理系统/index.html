一、项目概述

目标：实现一个基于控制台的“学生信息管理系统”，支持以下功能：

  • 添加学生
  • 删除学生
  • 修改学生信息
  • 查询所有学生
  • 按学号查询学生

技术点：

  • Java 基础语法
  • 类和对象
  • ArrayList 集合
  • Scanner 输入
  • 简单的控制台交互逻辑

二、项目结构

StudentManager/
├── Student.java           // 学生实体类
├── StudentManager.java    // 学生管理类
└── Main.java              // 程序入口类

三、核心类设计（含详细注释）

1. Student.java（学生实体类）

/**
 * 学生实体类，用于封装学生的基本信息
 */
public class Student {
    private String id;      // 学号
    private String name;    // 姓名
    private int age;        // 年龄
    private double score;   // 成绩

    /**
     * 构造方法，用于创建学生对象
     * @param id    学号
     * @param name  姓名
     * @param age   年龄
     * @param score 成绩
     */
    public Student(String id, String name, int age, double score) {
        this.id = id;
        this.name = name;
        this.age = age;
        this.score = score;
    }

    // 当时不知道编辑器都有自动生成的功能写这个是真烦
    // lombok确实是好东西
    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public double getScore() {
        return score;
    }

    public void setScore(double score) {
        this.score = score;
    }

    /**
     * 重写 toString 方法，方便打印学生信息
     */
    @Override
    public String toString() {
        return "学号: " + id + ", 姓名: " + name + ", 年龄: " + age + ", 成绩: " + score;
    }
}

2. StudentManager.java（学生管理类）

import java.util.ArrayList;

/**
 * 学生管理类，负责对学生信息进行增删改查操作
 */
public class StudentManager {
    // 使用 ArrayList 存储学生对象
    private ArrayList<Student> students = new ArrayList<>();

    /**
     * 添加学生
     * @param s 学生对象
     */
    public void addStudent(Student s) {
        students.add(s);
    }

    /**
     * 根据学号删除学生
     * @param id 学号
     * @return 删除成功返回 true，失败返回 false
     */
    public boolean deleteStudent(String id) {
        for (Student s : students) {
            if (s.getId().equals(id)) {
                students.remove(s);
                return true;
            }
        }
        return false;
    }

    /**
     * 根据学号查找学生
     * @param id 学号
     * @return 找到的学生对象，未找到返回 null
     */
    public Student findStudentById(String id) {
        for (Student s : students) {
            if (s.getId().equals(id)) {
                return s;
            }
        }
        return null;
    }

    /**
     * 修改学生信息
     * @param id    学号
     * @param name  新姓名
     * @param age   新年龄
     * @param score 新成绩
     */
    public void updateStudent(String id, String name, int age, double score) {
        Student s = findStudentById(id);
        if (s != null) {
            s.setName(name);
            s.setAge(age);
            s.setScore(score);
        }
    }

    /**
     * 打印所有学生信息
     */
    public void listAllStudents() {
        for (Student s : students) {
            System.out.println(s);
        }
    }
}

3. Main.java（程序入口类）

import java.util.Scanner;

/**
 * 程序入口类，提供控制台菜单交互
 */
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        StudentManager manager = new StudentManager();

        // 主循环，持续显示菜单
        while (true) {
            System.out.println("=== 学生管理系统 ===");
            System.out.println("1. 添加学生");
            System.out.println("2. 删除学生");
            System.out.println("3. 修改学生");
            System.out.println("4. 查询所有学生");
            System.out.println("5. 按学号查询");
            System.out.println("0. 退出");
            System.out.print("请选择操作：");

            int choice = scanner.nextInt();
            scanner.nextLine(); // 清除换行符

            switch (choice) {
                case 1:
                    // 添加学生
                    System.out.print("学号：");
                    String id = scanner.nextLine();
                    System.out.print("姓名：");
                    String name = scanner.nextLine();
                    System.out.print("年龄：");
                    int age = scanner.nextInt();
                    System.out.print("成绩：");
                    double score = scanner.nextDouble();
                    manager.addStudent(new Student(id, name, age, score));
                    System.out.println("学生添加成功！");
                    break;

                case 2:
                    // 删除学生
                    System.out.print("请输入要删除的学号：");
                    String delId = scanner.nextLine();
                    boolean deleted = manager.deleteStudent(delId);
                    System.out.println(deleted ? "删除成功" : "未找到该学生");
                    break;

                case 3:
                    // 修改学生
                    System.out.print("请输入要修改的学号：");
                    String updateId = scanner.nextLine();
                    System.out.print("新姓名：");
                    String newName = scanner.nextLine();
                    System.out.print("新年龄：");
                    int newAge = scanner.nextInt();
                    System.out.print("新成绩：");
                    double newScore = scanner.nextDouble();
                    manager.updateStudent(updateId, newName, newAge, newScore);
                    System.out.println("学生信息修改完成");
                    break;

                case 4:
                    // 查询所有学生
                    System.out.println("所有学生信息：");
                    manager.listAllStudents();
                    break;

                case 5:
                    // 按学号查询
                    System.out.print("请输入要查询的学号：");
                    String queryId = scanner.nextLine();
                    Student s = manager.findStudentById(queryId);
                    System.out.println(s != null ? s : "未找到该学生");
                    break;

                case 0:
                    // 退出程序
                    System.out.println("退出系统。");
                    return;

                default:
                    System.out.println("无效选择，请重试。");
            }
        }
    }
}
